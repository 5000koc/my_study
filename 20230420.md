# while 반복문
"""
while 조건 :
    반복할 조건 
"""
- 조건이 참(True)인 경우에 코드를 계속 반복

예제
- n = 1
- while n <= 10:
-     print(n)
-     n += 1 # n = n+1 과 같은 의미

- += 연산자
- 더하기 연산 후 할당
- n += 1 은 n = n+1 이라는 의미
- -= *= /= **= //= %= 산술연산자 다른 것들 모두 가능

- s1 = "안녕"
- s1 += "하세요" #문자로도 가능

# while 반복문을 활용하여 10부터 1까지 출력

- n = 10
- while n > 0: # while n >= 1: 과 같음
-     print(n)
-     n -= 1

- money = 10000
- price = 1000
- while money >= price:
-    print("커피를 구매했습니다")
-    money -= price
